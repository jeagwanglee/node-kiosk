npx sequelize model:generate --name item --attributes name:string,option_id:integer,price:integer,type:enum,amount:integer

npx sequelize model:generate --name option --attributes extra_price:integer,shot_price:integer,hot:boolean

npx sequelize model:generate --name order_item --attributes item_id:integer,amount:integer

npx sequelize model:generate --name item_order_customer --attributes item_id:integer,order_customer_id:integer,amount:integer

npx sequelize model:generate --name order_customer --attributes state:boolean
